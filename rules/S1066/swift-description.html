Noncompliant Code Example

  if condition2 {
    doSomething()
  }
}

if let y = someOptional {
  if x > 0 {
    doSomething()
  }
}

Compliant Solution

  doSomething()
}

  doSomething()
}

if x > 0, let y = someOptional {
