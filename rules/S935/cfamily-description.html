Noncompliant Code Example

int my_func (int a)
{
  if (a > 100)
  {
    return; // Noncompliant
  }

  if (a > 80)
  {
    throw new Exception(); // Compliant
  }

  // Noncompliant
}

Compliant Solution

int my_func (int a)
{
  if (a > 100)
  {
    return 12;
  }

  if (a > 80)
  {
    throw new Exception();
  }

  return a;
}
