Using the <code>this</code> keyword inside the scope of an object to refer to the objectâ€™s properties and methods yields cleaner, clearer code, and helps avoid confusion when there are variables or functions outside the object scope with the same or similar names.

Noncompliant Code Example

function Person(name, birthdate) {
  this.name = name;
  this.birthdate = birthdate;

  get name() {
    return name; // Noncompliant
  }
}

Compliant Solution

function Person(name, birthdate) {
  this.name = name;
  this.birthdate = birthdate;

  get name() {
    return this.name;
  }
}
