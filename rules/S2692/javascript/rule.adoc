Most checks against an <code>indexOf</code> call against a string or array compare it with -1 because 0 is a valid index. Any checks which look for values >0 ignore the first element, which is likely a bug. If you're merely checking the presence of the string, consider using <code>includes</code> instead. Before usingÂ <code>includes</code> method make sure that your browser version is supporting it.

== Noncompliant Code Example

----
var color = "blue";
var name = "ishmael";
var number = 123;

var arr = [color, name];

if (arr.indexOf("blue") > 0) { // Noncompliant
  // ...
}
if (name.indexOf("ish") > 0) { // Noncompliant
  // ...
}
----

== Compliant Solution

----
var color = "blue";
var name = "ishmael";
var number = 123;

var arr = [color, name];

if (arr.indexOf("blue") >= 0) { 
  // ...
}
if (name.includes("ish")) { 
  // ...
}
----
